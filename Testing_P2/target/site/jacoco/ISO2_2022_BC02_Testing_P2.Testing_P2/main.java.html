package ISO2_2022_BC02_Testing_P2.Testing_P2;

public class main {
	public static void main(String[]args) throws numCero, numNegativo, anguloNoValido{
		Triangulo t = null;
		try {
			t = new Triangulo("11", "12", "12", "181", "90", "90");
		}catch (numCero numCero) {
			System.out.println(numCero.getMessage());
			System.exit(0);
		}catch (numNegativo numNegativo) {
			System.out.println(numNegativo.getMessage());
			System.exit(0);
		}catch (anguloNoValido anguloNoValido) {
			System.out.println(anguloNoValido.getMessage());
			System.exit(0);
		}
		try {
			t.setL1("0");
		}catch(numCero e) {
			System.out.println(e.getMessage());
		}
		
		System.out.println("Triángulo correcto");
		System.out.println("Según ángulos es un triángulo: " + t.clasificarPorAngulos(t.getA1(), t.getA2(), t.getA3()));
		System.out.println("Segun lados es un triángulo: "+ t.clasificarPorLados(t.getL1(), t.getL2(), t.getL3()));
		
	}
}
